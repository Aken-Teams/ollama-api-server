    API 使用文檔
    

    
    
    
        
    Base URL：https://ollama_pjapi.theaken.com
    
        
    認證方式：Authorization: Bearer YOUR_API_KEY
    
        
    所有 API 均相容 OpenAI 格式，可直接替換 base_url 使用。
    
    
    

    
    
        
        
    
            
    import requests, json

API_URL = "https://ollama_pjapi.theaken.com"
API_KEY = "YOUR_API_KEY"
headers = {"Authorization": f"Bearer {API_KEY}", "Content-Type": "application/json"}
    
        
    
    
    

    
    
    Chat Completions API
    
    
    POST /v1/chat/completions — 對話生成，支援串流與非串流模式。
    

    
    可用模型
    
    
    
        
    類型提供者模型說明
    
        
            
    
                地端
                llama.cpp
                gpt-oss:120b
                120B 通用大模型，複雜推理
            
    
            
    
                地端
                llama.cpp
                gemma4:31b
                Google Gemma4 31B
            
    
            
    
                地端
                llama.cpp
                Qwen3.5:122b
                通義千問 MoE，256K 上下文
            
    
            
    
                地端
                Ollama
                qwen2.5:72b
                通義千問 72B，中文頂尖
            
    
            
    
                地端
                Ollama
                llava:7b
                視覺模型，支援圖片輸入
            
    
            
    
                雲端
                DeepSeek
                deepseek-chat
                DeepSeek 對話模型
            
    
            
    
                雲端
                DeepSeek
                deepseek-reasoner
                DeepSeek 推理模型，數學/邏輯
            
    
            
    
    

    
    對話範例
    
    
    
        
        
    
            
    # 基本對話
response = requests.post(f"{API_URL}/v1/chat/completions", headers=headers, json={
    "model": "gpt-oss:120b",
    "messages": [
        {"role": "system", "content": "你是一個專業的助手"},
        {"role": "user", "content": "請解釋什麼是機器學習"}
    ]
})
print(response.json()["choices"][0]["message"]["content"])
    
        
    
    
    

    
    串流回應
    
    
    
        
        
    
            
    # 串流模式 — 逐字輸出
response = requests.post(f"{API_URL}/v1/chat/completions", headers=headers, json={
    "model": "deepseek-chat",
    "messages": [{"role": "user", "content": "寫一篇關於 AI 的短文"}],
    "stream": True
}, stream=True)

for line in response.iter_lines():
    if line:
        line = line.decode("utf-8")
        if line.startswith("data: ") and line != "data: [DONE]":
            data = json.loads(line[6:])
            delta = data["choices"][0].get("delta", {})
            if "content" in delta:
                print(delta["content"], end="", flush=True)

    OCR 文件辨識 API
    
    
    POST /v1/ocr/recognize — 支援 JPG/PNG/GIF/WebP/BMP/PDF，上限 20MB。
    

    
    
        
    模型 ID說明最適場景
    
        
            
    deepseek-ocrGPU 加速、高精度、智能版面複雜文件、手寫、多語言
    
            
    pp-ocrv5PaddleOCR v5、80+ 語言高精度、表格辨識
    
            
    llava-ocrAI 視覺模型、理解上下文手寫辨識、複雜版面
    
            
    general-ocr通用高精度 OCR印刷文件、清晰圖片
    
            
    table-ocr表格結構保留表格、報表
    
            
    invoice-ocr發票格式辨識發票、收據、帳單
    
            
    
    

    
    
        
        
    
            
    # OCR 辨識
with open("document.png", "rb") as f:
    response = requests.post(f"{API_URL}/v1/ocr/recognize",
        headers={"Authorization": f"Bearer {API_KEY}"},
        files={"file": f},
        data={"model": "deepseek-ocr", "output_format": "text", "language": "auto"})
print(response.json()["text"])

# PDF 辨識（自動偵測內嵌文字或轉圖片 OCR）
with open("report.pdf", "rb") as f:
    response = requests.post(f"{API_URL}/v1/ocr/recognize",
        headers={"Authorization": f"Bearer {API_KEY}"},
        files={"file": f},
        data={"model": "pp-ocrv5", "output_format": "markdown"})
print(response.json()["text"])
