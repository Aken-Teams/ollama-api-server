    📝 程式碼範例
    

            
    Python 完整範例
    
            
    
                
                
    
                    
    import requests
import json

# API 設定
API_URL = "https://ollama_pjapi.theaken.com"

# 1. 列出可用模型
response = requests.get(f"{API_URL}/v1/models")
models = response.json()
print(f"可用模型: {[m['id'] for m in models['data']]}")

# 2. 聊天完成請求
chat_request = {
    "model": "qwen2.5:3b",
    "messages": [
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user", "content": "Hello, how are you?"}
    ],
    "temperature": 0.7
}

response = requests.post(
    f"{API_URL}/v1/chat/completions",
    json=chat_request
)

if response.status_code == 200:
    result = response.json()
    print(f"AI 回應: {result['choices'][0]['message']['content']}")

# 3. 串流模式
chat_request["stream"] = True
response = requests.post(
    f"{API_URL}/v1/chat/completions",
    json=chat_request,
    stream=True
)

for line in response.iter_lines():
    if line:
        if line.startswith(b"data: "):
            data_str = line[6:].decode('utf-8')
            if data_str.strip() != "[DONE]":
                data = json.loads(data_str)
                if 'choices' in data:
                    delta = data['choices'][0].get('delta', {})
                    if 'content' in delta:
                        print(delta['content'], end='', flush=True)
    
                
    
            
    
            
            
    JavaScript/Node.js 範例
    
            
    
                
                
    
                    
    // 使用 fetch API
const API_URL = 'https://ollama_pjapi.theaken.com';

// 1. 列出模型
async function listModels() {
    const response = await authFetch(`${API_URL}/v1/models`);
    const models = await response.json();
    console.log('可用模型:', models.data.map(m => m.id));
}

// 2. 聊天請求
async function chat(message) {
    const response = await authFetch(`${API_URL}/v1/chat/completions`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            model: 'qwen2.5:3b',
            messages: [
                { role: 'system', content: 'You are a helpful assistant.' },
                { role: 'user', content: message }
            ],
            temperature: 0.7
        })
    });
    
    const result = await response.json();
    return result.choices[0].message.content;
}

// 3. 串流模式
async function chatStream(message) {
    const response = await authFetch(`${API_URL}/v1/chat/completions`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({
            model: 'qwen2.5:3b',
            messages: [
                { role: 'user', content: message }
            ],
            stream: true
        })
    });
    
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    
    while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        const lines = chunk.split('\\n');
        
        for (const line of lines) {
            if (line.startsWith('data: ')) {
                const data = line.slice(6);
                if (data.trim() === '[DONE]') continue;
                
                try {
                    const json = JSON.parse(data);
                    if (json.choices?.[0]?.delta?.content) {
                        process.stdout.write(json.choices[0].delta.content);
                    }
                } catch (e) {}
            }
        }
    }
}
